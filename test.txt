  def parse_filenames(files)
    parsed = files.map do |file|
      file[21.. file.length]
    end
    return parsed
  end


  def self.new_by_filename(name)
    file_name = name.split(' - ')
    song = Song.new(file_name[1])
    artist = Artist.find_or_create_by_name(file_name[0])
    song.artist = artist
    return song
  end
